                const relevantProperties = properties.filter(p => p.category === activePrimaryFilter);

                source = [...new Set(relevantProperties.flatMap(p => p.features))];

                buildFunc = (key) => featureDetails[key];

            } else {

                return; // No secondary filters for "All"

            }

            

            if (source.length > 0) {

                const allButton = document.createElement('button');

                allButton.className = 'border-2 border-rose-500 bg-rose-500 text-white font-semibold py-2 px-4 rounded-full transition';

                allButton.dataset.feature = 'all';

                allButton.innerHTML = `All`;

                secondaryFiltersDiv.appendChild(allButton);


                source.forEach(item => {

                    const details = buildFunc(item);

                    if (details) {

                        const button = document.createElement('button');

                        button.className = 'border font-semibold py-2 px-4 rounded-full hover:border-gray-900 transition';

                        button.dataset.feature = item;

                        button.innerHTML = `<i class="${details.icon} mr-2"></i>${details.name}`;

                        secondaryFiltersDiv.appendChild(button);

                    }

                });

                secondaryFiltersContainer.classList.add('active');

            }


            secondaryFiltersDiv.querySelectorAll('button').forEach(button => {

                button.addEventListener('click', () => {

                    activeSecondaryFilter = button.dataset.feature;

                    secondaryFiltersDiv.querySelectorAll('button').forEach(btn => btn.className = 'border font-semibold py-2 px-4 rounded-full hover:border-gray-900 transition');

                    button.className = 'border-2 border-rose-500 bg-rose-500 text-white font-semibold py-2 px-4 rounded-full transition';

                    renderProperties();

                });

            });

        }

        function setupDropZone(dropZoneId, previewContainerId, placeholderText) {

            const dropZone = document.getElementById(dropZoneId);

            if (!dropZone) return;


            dropZone.addEventListener('click', () => {

                const container = document.getElementById(previewContainerId);

                const item = document.createElement('div');

                item.className = 'preview-item';

                item.innerHTML = `<img src="https://placehold.co/100x100/e0e7ff/333333?text=${placeholderText}" alt="Preview">`;

                

                if (previewContainerId.includes('layout')) {

                    container.innerHTML = '';

                }

                container.appendChild(item);

            });

        }
